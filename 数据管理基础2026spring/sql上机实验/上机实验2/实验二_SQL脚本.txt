/*  第一部分：数据库创建  */
use mysql;
drop schema if exists myschool26; # 删除原有的数据模式
create schema myschool26 charset gb2312 collate gb2312_bin;
use myschool26;# 设置当前工作模式

#创建学生表 students
CREATE TABLE students
(
    sno        CHAR(9) PRIMARY KEY,
    sname      CHAR(20) NOT NULL,
    ssex       CHAR(2)  NOT NULL CHECK (ssex IN ('男', '女')),
    birthday   DATE     NOT NULL,
    dept       CHAR(20),
    enrolldate DATE
);

#创建教师表 teachers
CREATE TABLE teachers
(
    tno   CHAR(9) PRIMARY KEY,
    tname CHAR(20) NOT NULL,
    dept  CHAR(20)
);

#创建课程表 courses
CREATE TABLE courses
(
    cno    CHAR(8) PRIMARY KEY,
    cname  CHAR(40) NOT NULL,
    credit INT      NOT NULL CHECK (credit > 0 AND credit < 10),
    chours INT      NOT NULL,
    dept   CHAR(20)
);

#创建课程班表 courseclass
CREATE TABLE courseclass
(
    clsno   CHAR(8) PRIMARY KEY,
    cno     CHAR(8) NOT NULL,
    tno     CHAR(9),
    clsyear INT     NOT NULL,
    clsterm CHAR(4) NOT NULL CHECK (clsterm IN ('春季', '秋季', '暑期')),
    FOREIGN KEY (cno) REFERENCES courses (cno) ON UPDATE CASCADE ON DELETE RESTRICT,
    FOREIGN KEY (tno) REFERENCES teachers (tno) ON UPDATE CASCADE ON DELETE SET NULL
);

#创建选课表 enrollment
CREATE TABLE enrollment
(
    sno   CHAR(9) NOT NULL,
    clsno CHAR(8) NOT NULL,
    grade INT CHECK (grade >= 0 AND grade <= 100),
    PRIMARY KEY (sno, clsno),
    FOREIGN KEY (sno) REFERENCES students (sno) ON UPDATE CASCADE ON DELETE RESTRICT,
    FOREIGN KEY (clsno) REFERENCES courseclass (clsno) ON UPDATE CASCADE ON DELETE RESTRICT
);

/*  第三部分：数据查询  */
#1.查询满足下述条件的课程的课程号、课程名、开课院系：课程名中含有汉字‘数’；结果按照开课院系和课程号的升序排列。
SELECT cno, cname, dept
FROM courses
WHERE cname LIKE '%数%'
ORDER BY dept, cno;

#2.查询满足下述条件的学生的学号、姓名、就读院系：姓名中的第二个汉字是‘静’；结果按照学生姓名的升序排列。
SELECT sno, sname, dept
FROM students
WHERE sname LIKE '_静%'
ORDER BY sname;

#3.查询成绩为空值的选课记录，结果返回选课学生的学号和姓名、课程的课程号和课程名；结果按照课程号和学号的升序排列。
SELECT students.sno, students.sname, courses.cno, courses.cname
FROM enrollment
         JOIN students ON enrollment.sno = students.sno
         JOIN courseclass ON enrollment.clsno = courseclass.clsno
         JOIN courses ON courseclass.cno = courses.cno
WHERE enrollment.grade IS NULL
ORDER BY courses.cno, students.sno;

#4.查询下列学生的课程修读情况（不考虑没有成绩的课程）：2023级、且所有成绩都及格；结果返回满足条件学生的学号，姓名，修读课程的门数、学分总数及平均成绩；并按照学号的升序输出查询结果。
SELECT s.sno         AS 学号,
       s.sname       AS 姓名,
       COUNT(c.cno)  AS 修读门数,
       SUM(c.credit) AS 学分总数,
       AVG(e.grade)  AS 平均成绩
FROM students s
         JOIN enrollment e ON s.sno = e.sno
         JOIN courseclass cc ON e.clsno = cc.clsno
         JOIN courses c ON cc.cno = c.cno
WHERE e.grade IS NOT NULL
  AND YEAR(s.enrolldate) = 2023
GROUP BY s.sno
HAVING MIN(e.grade) >= 60 
ORDER BY s.sno;

#5.查询承担过授课任务的教师的授课情况，结果返回教师的工号和姓名、授课次数（讲授的课程班数）、累计课时数。结果按照累计课时数的降序和教师工号的升序排列。
SELECT t.tno           AS 教师工号,
       t.tname         AS 姓名,
       COUNT(cc.clsno) AS 授课次数,
       SUM(c.chours)   AS 累计课时数
FROM teachers t
         JOIN courseclass cc ON t.tno = cc.tno
         JOIN courses c ON cc.cno = c.cno
GROUP BY t.tno
ORDER BY 累计课时数 DESC, t.tno;

#6.按照姓氏统计所有在校学生的人数，结果返回姓氏和学生人数；结果按照人数的降序和姓氏的升序排列。说明：所有学生均为单字的姓氏，不考虑复姓情况；需要使用系统内置的字符处理函数。
SELECT LEFT(sname, 1) AS 姓氏,
       COUNT(sno)     AS 学生人数
FROM students
GROUP BY LEFT(sname, 1)
ORDER BY 学生人数 DESC, 姓氏;

#7.满足下述条件的学生的就读院系、学号和姓名：2021级、且存在自己就读学院开设的课程没有通过（没有选修、或没有成绩、或成绩不及格），结果按照学生就读院系和学号的升序排列。
SELECT s.dept, s.sno, s.sname
FROM students s
WHERE YEAR(s.enrolldate) = 2021
  AND (SELECT COUNT(DISTINCT cc.cno)
       FROM enrollment e
                JOIN courseclass cc ON e.clsno = cc.clsno
                JOIN courses c ON cc.cno = c.cno
       WHERE e.sno = s.sno
         AND c.dept = s.dept
         AND e.grade >= 60) < (SELECT COUNT(*)
                               FROM courses c2
                               WHERE c2.dept = s.dept)
ORDER BY s.dept, s.sno;

#8.满足下述条件的学生的就读院系、学号和姓名：2021级、且选修通过了自己就读学院开设的所有课程，结果按照学生就读院系和学号的升序排列。
SELECT s.dept, s.sno, s.sname
FROM students s
WHERE YEAR(s.enrolldate) = 2021
  AND (SELECT COUNT(DISTINCT cc.cno)
       FROM enrollment e
                JOIN courseclass cc ON e.clsno = cc.clsno
                JOIN courses c ON cc.cno = c.cno
       WHERE e.sno = s.sno
         AND c.dept = s.dept
         AND e.grade >= 60) = (SELECT COUNT(*)
                               FROM courses c2
                               WHERE c2.dept = s.dept)
ORDER BY s.dept, s.sno;

#9.查询每一位教师的授课起止年份，结果返回教师的工号、姓名、工作院系、第一次授课的年份、最后一次授课的年份；如果某位老师没有承担过授课任务，则授课起止年份返回空值；结果按照工作院系和教师工号的升序排列。
SELECT t.tno           AS 教师工号,
       t.tname         AS 姓名,
       t.dept          AS 工作院系,
       MIN(cc.clsyear) AS 第一次授课年份,
       MAX(cc.clsyear) AS 最后一次授课年份
FROM teachers t
         LEFT JOIN courseclass cc ON t.tno = cc.tno
GROUP BY t.tno, t.tname, t.dept
ORDER BY t.dept, t.tno;

#10.在一门课程的各个课程班中，查询各课程班的班平均成绩的最高值；结果返回课程的课程号、课程名、班平均成绩最高的课程班的代码、授课年份和授课学期；并按照课程号的升序和授课年份的降序输出查询结果。（在计算班平均成绩时，不需要考虑成绩为空值的选课元组）
SELECT c.cno      AS 课程号,
       c.cname    AS 课程名,
       cc.clsno   AS 课程班代码,
       cc.clsyear AS 授课年份,
       cc.clsterm AS 授课学期
FROM courses c
         JOIN courseclass cc ON c.cno = cc.cno
         JOIN enrollment e ON cc.clsno = e.clsno
WHERE e.grade IS NOT NULL
GROUP BY cc.clsno, c.cno, cc.clsyear
HAVING AVG(e.grade) >= ALL (SELECT AVG(e2.grade)
                            FROM courseclass cc2
                                     JOIN enrollment e2 ON cc2.clsno = e2.clsno
                            WHERE e2.grade IS NOT NULL
                              AND cc2.cno = c.cno
                            GROUP BY cc2.clsno)
ORDER BY c.cno, cc.clsyear DESC;

/*  第五部分：视图创建与访问  */
#1.创建第一个视图：用于查询每一门课程的累计修读学生人数和该门课程的平均成绩（视图属性包括：课程号、累计修读学生人数、平均成绩）
CREATE VIEW v_course_statistics AS
SELECT
    cc.cno AS 课程号,
    COUNT(e.sno) AS 累计修读学生人数,
    AVG(e.grade) AS 平均成绩
FROM courseclass cc
    JOIN enrollment e ON cc.clsno = e.clsno
GROUP BY cc.cno;

#2.视图的嵌套定义和查询
#(1)创建第二个视图，用于查询每一门课程的累计修读学生人数、成绩大于课程平均成绩的学生人数（视图属性包括：课程号、累计修读学生人数、大于课程平均成绩的学生人数）
CREATE VIEW v_course_performance AS
SELECT
    v1.课程号,
    v1.累计修读学生人数,
    SUM(IF(e.grade > v1.平均成绩, 1, 0)) AS 大于课程平均成绩的学生人数
FROM v_course_statistics v1
         JOIN courseclass cc ON v1.课程号 = cc.cno
         JOIN enrollment e ON cc.clsno = e.clsno
GROUP BY v1.课程号, v1.累计修读学生人数;
#(2)查询满足下述条件的课程的课程号、课程名、开课院系：在修读过的学生中，半数以上学生的成绩低于该课程平均成绩。结果按照开课院系和课程号的升序排列。
SELECT
    c.cno AS 课程号,
    c.cname AS 课程名,
    c.dept AS 开课院系
FROM courses c
         JOIN v_course_statistics v1 ON c.cno = v1.课程号
         JOIN courseclass cc ON c.cno = cc.cno
         JOIN enrollment e ON cc.clsno = e.clsno
GROUP BY c.cno, c.cname, c.dept, v1.累计修读学生人数
HAVING SUM(IF(e.grade < v1.平均成绩, 1, 0)) > (v1.累计修读学生人数 / 2.0)
ORDER BY c.dept, c.cno;
